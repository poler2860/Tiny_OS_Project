54/*
55 The thread layout.
56 --------------------
57
58 On the x86 (Pentium) architecture, the stack grows upward. Therefore, we
59 can allocate the TCB at the top of the memory block used as the stack.
60
61 +-------------+
62 | TCB |
63 +-------------+
64 | |
65 | stack |
66 | |
67 | ^ |
68 | | |
69 +-------------+
70 | first frame |
71 +-------------+
72
73 Advantages: (a) unified memory area for stack and TCB (b) stack overrun will
74 crash own thread, before it affects other threads (which may make debugging
75 easier).
76
77 Disadvantages: The stack cannot grow unless we move the whole TCB. Of course,
78 we do not support stack growth anyway!
79 */
80
81/*
82 A counter for active threads. By "active", we mean 'existing',
83 with the exception of idle threads (they don't count).
84 */
85volatile unsigned int active_threads = 0;
86Mutex active_threads_spinlock = MUTEX_INIT;
87
88/* This is specific to Intel Pentium! */
89#define SYSTEM_PAGE_SIZE (1 << 12)
90
91/* The memory allocated for the TCB must be a multiple of SYSTEM_PAGE_SIZE */
92#define THREAD_TCB_SIZE \
93 (((sizeof(TCB) + SYSTEM_PAGE_SIZE - 1) / SYSTEM_PAGE_SIZE) * SYSTEM_PAGE_SIZE)
94
95#define THREAD_SIZE (THREAD_TCB_SIZE + THREAD_STACK_SIZE)

149/*
150 Initialize and return a new TCB
151*/
152
153TCB* spawn_thread(PCB* pcb, void (*func)())
154{
155 /* The allocated thread size must be a multiple of page size */
156 TCB* tcb = (TCB*)allocate_thread(THREAD_SIZE);
157
158 /* Set the owner */
159 tcb->owner_pcb = pcb;
160
161 /* Initialize the other attributes */
162 tcb->type = NORMAL_THREAD;
163 tcb->state = INIT;
164 tcb->phase = CTX_CLEAN;
165 tcb->thread_func = func;
166 tcb->wakeup_time = NO_TIMEOUT;
167 rlnode_init(&tcb->sched_node, tcb); /* Intrusive list node */
168
169 tcb->its = QUANTUM;
170 tcb->rts = QUANTUM;
171 tcb->last_cause = SCHED_IDLE;
172 tcb->curr_cause = SCHED_IDLE;
173
174 /* Compute the stack segment address and size */
175 void* sp = ((void*)tcb) + THREAD_TCB_SIZE;
176
177 /* Init the context */
178 cpu_initialize_context(&tcb->context, sp, THREAD_STACK_SIZE, thread_start);
179
180#ifndef NVALGRIND
181 tcb->valgrind_stack_id = VALGRIND_STACK_REGISTER(sp, sp + THREAD_STACK_SIZE);
182#endif
183
184 /* increase the count of active threads */
185 Mutex_Lock(&active_threads_spinlock);
186 active_threads++;
187 Mutex_Unlock(&active_threads_spinlock);
188
189 return tcb;
190}

218/*
219 The scheduler queue is implemented as a doubly linked list. The
220 head and tail of this list are stored in SCHED.
221
222 Also, the scheduler contains a linked list of all the sleeping
223 threads with a timeout.
224
225 Both of these structures are protected by @c sched_spinlock.
226*/
227
228rlnode SCHED; /* The scheduler queue */
229rlnode TIMEOUT_LIST; /* The list of threads with a timeout */
230Mutex sched_spinlock = MUTEX_INIT; /* spinlock for scheduler queue */

240/*
241 Possibly add TCB to the scheduler timeout list.
242
243 *** MUST BE CALLED WITH sched_spinlock HELD ***
244*/
245static void sched_register_timeout(TCB* tcb, TimerDuration timeout)
246{
247 if (timeout != NO_TIMEOUT) {
248 /* set the wakeup time */
249 TimerDuration curtime = bios_clock();
250 tcb->wakeup_time = (timeout == NO_TIMEOUT) ? NO_TIMEOUT : curtime + timeout;
251
252 /* add to the TIMEOUT_LIST in sorted order */
253 rlnode* n = TIMEOUT_LIST.next;
254 for (; n != &TIMEOUT_LIST; n = n->next)
255 /* skip earlier entries */
256 if (tcb->wakeup_time < n->tcb->wakeup_time)
257 break;
258 /* insert before n */
259 rl_splice(n->prev, &tcb->sched_node);
260 }
261}
262
263/*
264 Add TCB to the end of the scheduler list.
265
266 *** MUST BE CALLED WITH sched_spinlock HELD ***
267*/
268static void sched_queue_add(TCB* tcb)
269{
270 /* Insert at the end of the scheduling list */
271 rlist_push_back(&SCHED, &tcb->sched_node);
272
273 /* Restart possibly halted cores */
274 cpu_core_restart_one();
275}

277/*
278 Adjust the state of a thread to make it READY.
279
280 *** MUST BE CALLED WITH sched_spinlock HELD ***
281 */
282static void sched_make_ready(TCB* tcb)
283{
284 assert(tcb->state == STOPPED || tcb->state == INIT);
285
286 /* Possibly remove from TIMEOUT_LIST */
287 if (tcb->wakeup_time != NO_TIMEOUT) {
288 /* tcb is in TIMEOUT_LIST, fix it */
289 assert(tcb->sched_node.next != &(tcb->sched_node) && tcb->state == STOPPED);
290 rlist_remove(&tcb->sched_node);
291 tcb->wakeup_time = NO_TIMEOUT;
292 }
293
294 /* Mark as ready */
295 tcb->state = READY;
296
297 /* Possibly add to the scheduler queue */
298 if (tcb->phase == CTX_CLEAN)
299 sched_queue_add(tcb);
300}
301
302/*
303 Scan the \c TIMEOUT_LIST for threads whose timeout has expired, and
304 wake them up.
305
306 *** MUST BE CALLED WITH sched_spinlock HELD ***
307*/
308static void sched_wakeup_expired_timeouts()
309{
310 /* Empty the timeout list up to the current time and wake up each thread */
311 TimerDuration curtime = bios_clock();
312
313 while (!is_rlist_empty(&TIMEOUT_LIST)) {
314 TCB* tcb = TIMEOUT_LIST.next->tcb;
315 if (tcb->wakeup_time > curtime)
316 break;
317 sched_make_ready(tcb);
318 }
319}
320
321/*
322 Remove the head of the scheduler list, if any, and
323 return it. Return NULL if the list is empty.
324
325 *** MUST BE CALLED WITH sched_spinlock HELD ***
326*/
327static TCB* sched_queue_select(TCB* current)
328{
329 /* Get the head of the SCHED list */
330 rlnode* sel = rlist_pop_front(&SCHED);
331
332 TCB* next_thread = sel->tcb; /* When the list is empty, this is NULL */
333
334 if (next_thread == NULL)
335 next_thread = (current->state == READY) ? current : &CURCORE.idle_thread;
336
337 next_thread->its = QUANTUM;
338
339 return next_thread;
340}

342/*
343 Make the process ready.
344 */
345int wakeup(TCB* tcb)
346{
347 int ret = 0;
348
349 /* Preemption off */
350 int oldpre = preempt_off;
351
352 /* To touch tcb->state, we must get the spinlock. */
353 Mutex_Lock(&sched_spinlock);
354
355 if (tcb->state == STOPPED || tcb->state == INIT) {
356 sched_make_ready(tcb);
357 ret = 1;
358 }
359
360 Mutex_Unlock(&sched_spinlock);
361
362 /* Restore preemption state */
363 if (oldpre)
364 preempt_on;
365
366 return ret;
367}

369/*
370 Atomically put the current process to sleep, after unlocking mx.
371 */
372void sleep_releasing(Thread_state state, Mutex* mx, enum SCHED_CAUSE cause,
373 TimerDuration timeout)
374{
375 assert(state == STOPPED || state == EXITED);
376
377
378 int preempt = preempt_off;
379 TCB* tcb = CURTHREAD;
380 Mutex_Lock(&sched_spinlock);
381
382 /* mark the thread as stopped or exited */
383 tcb->state = state;
384
385 /* register the timeout (if any) for the sleeping thread */
386 if (state != EXITED)
387 sched_register_timeout(tcb, timeout);
388
389 /* Release mx */
390 if (mx != NULL)
391 Mutex_Unlock(mx);
392
393 /* Release the schduler spinlock before calling yield() !!! */
394 Mutex_Unlock(&sched_spinlock);
395
396 /* call this to schedule someone else */
397 yield(cause);
398
399 /* Restore preemption state */
400 if (preempt)
401 preempt_on;
402}

404/* This function is the entry point to the scheduler's context switching */
405
406void yield(enum SCHED_CAUSE cause)
407{
408 /* Reset the timer, so that we are not interrupted by ALARM */
409 TimerDuration remaining = bios_cancel_timer();
410
411 /* We must stop preemption but save it! */
412 int preempt = preempt_off;
413
414 TCB* current = CURTHREAD; /* Make a local copy of current process, for speed */
415
416 Mutex_Lock(&sched_spinlock);
417
418 /* Update CURTHREAD state */
419 if (current->state == RUNNING)
420 current->state = READY;
421
422 /* Update CURTHREAD scheduler data */
423 current->rts = remaining;
424 current->last_cause = current->curr_cause;
425 current->curr_cause = cause;
426
427 /* Wake up threads whose sleep timeout has expired */
428 sched_wakeup_expired_timeouts();
429
430 /* Get next */
431 TCB* next = sched_queue_select(current);
432 assert(next != NULL);
433
434 /* Save the current TCB for the gain phase */
435 CURCORE.previous_thread = current;
436
437 Mutex_Unlock(&sched_spinlock);
438
439 /* Switch contexts */
440 if (current != next) {
441 CURTHREAD = next;
442 cpu_swap_context(&current->context, &next->context);
443 }
444
445 /* This is where we get after we are switched back on! A long time
446 may have passed. Start a new timeslice...
447 */
448 gain(preempt);
449}
450
451/*
452 This function must be called at the beginning of each new timeslice.
453 This is done mostly from inside yield().
454 However, for threads that are executed for the first time, this
455 has to happen in thread_start.
456
457 The 'preempt' argument determines whether preemption is turned on
458 in the new timeslice. When returning to threads in the non-preemptive
459 domain (e.g., waiting at some driver), we need to not turn preemption
460 on!
461*/
462
463void gain(int preempt)
464{
465 Mutex_Lock(&sched_spinlock);
466
467 TCB* current = CURTHREAD;
468
469 /* Mark current state */
470 current->state = RUNNING;
471 current->phase = CTX_DIRTY;
472 current->rts = current->its;
473
474 /* Take care of the previous thread */
475 TCB* prev = CURCORE.previous_thread;
476 if (current != prev) {
477 prev->phase = CTX_CLEAN;
478 switch (prev->state) {
479 case READY:
480 if (prev->type != IDLE_THREAD)
481 sched_queue_add(prev);
482 break;
483 case EXITED:
484 release_TCB(prev);
485 break;
486 case STOPPED:
487 break;
488 default:
489 assert(0); /* prev->state should not be INIT or RUNNING ! */
490 }
491 }
492
493 Mutex_Unlock(&sched_spinlock);
494
495 /* Reset preemption as needed */
496 if (preempt)
497 preempt_on;
498
499 /* Set a 1-quantum alarm */
500 bios_set_timer(current->rts);
501}

503static void idle_thread()
504{
505 /* When we first start the idle thread */
506 yield(SCHED_IDLE);
507
508 /* We come here whenever we cannot find a ready thread for our core */
509 while (active_threads > 0) {
510 cpu_core_halt();
511 yield(SCHED_IDLE);
512 }
513
514 /* If the idle thread exits here, we are leaving the scheduler! */
515 bios_cancel_timer();
516 cpu_core_restart_all();
517}

528void run_scheduler()
529{
530 CCB* curcore = &CURCORE;
531
532 /* Initialize current CCB */
533 curcore->id = cpu_core_id;
534
535 curcore->current_thread = &curcore->idle_thread;
536
537 curcore->idle_thread.owner_pcb = get_pcb(0);
538 curcore->idle_thread.type = IDLE_THREAD;
539 curcore->idle_thread.state = RUNNING;
540 curcore->idle_thread.phase = CTX_DIRTY;
541 curcore->idle_thread.wakeup_time = NO_TIMEOUT;
542 rlnode_init(&curcore->idle_thread.sched_node, &curcore->idle_thread);
543
544 curcore->idle_thread.its = QUANTUM;
545 curcore->idle_thread.rts = QUANTUM;
546
547 curcore->idle_thread.curr_cause = SCHED_IDLE;
548 curcore->idle_thread.last_cause = SCHED_IDLE;
549
550 /* Initialize interrupt handler */
551 cpu_interrupt_handler(ALARM, yield_handler);
552 cpu_interrupt_handler(ICI, ici_handler);
553
554 /* Run idle thread */
555 preempt_on;
556 idle_thread();
557
558 /* Finished scheduling */
559 assert(CURTHREAD == &CURCORE.idle_thread);
560 cpu_interrupt_handler(ALARM, NULL);
561 cpu_interrupt_handler(ICI, NULL);
562}
