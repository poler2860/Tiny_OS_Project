32/* Per-core boot function for tinyos */
33void boot_tinyos_kernel()
34{
35
36 if(cpu_core_id==0) {
37 /* Initialize the kenrel data structures */
38 initialize_processes();
39 initialize_devices();
40 initialize_files();
41 initialize_scheduler();
42
43 /* The boot task is executed normally! */
44 if(Exec(boot_rec.init_task, boot_rec.argl, boot_rec.args)!=1)
45 FATAL("The init process does not have PID==1");
46 }
47
48 cpu_core_barrier_sync();
49
50#ifndef NVALGRIND
51 VALGRIND_PRINTF_BACKTRACE("TINYOS: Entering scheduler for core %d\n",cpu_core_id);
52#endif
53
54 run_scheduler();
55
56 if(cpu_core_id==0) {
57 /* Here, we could add cleanup after the scheduler has ended. */
58 }
59}
