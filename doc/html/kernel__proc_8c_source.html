50static PCB* pcb_freelist;
51
52void initialize_processes()
53{
54 /* initialize the PCBs */
55 for(Pid_t p=0; p<MAX_PROC; p++) {
56 initialize_PCB(&PT[p]);
57 }
58
59 /* use the parent field to build a free list */
60 PCB* pcbiter;
61 pcb_freelist = NULL;
62 for(pcbiter = PT+MAX_PROC; pcbiter!=PT; ) {
63 --pcbiter;
64 pcbiter->parent = pcb_freelist;
65 pcb_freelist = pcbiter;
66 }
67
68 process_count = 0;
69
70 /* Execute a null "idle" process */
71 if(Exec(NULL,0,NULL)!=0)
72 FATAL("The scheduler process does not have pid==0");
73}

76/*
77 Must be called with kernel_mutex held
78*/
79PCB* acquire_PCB()
80{
81 PCB* pcb = NULL;
82
83 if(pcb_freelist != NULL) {
84 pcb = pcb_freelist;
85 pcb->pstate = ALIVE;
86 pcb_freelist = pcb_freelist->parent;
87 process_count++;
88 }
89
90 return pcb;
91}

96void release_PCB(PCB* pcb)
97{
98 pcb->pstate = FREE;
99 pcb->parent = pcb_freelist;
100 pcb_freelist = pcb;

128/*
129 System call to create a new process.
130 */
131Pid_t sys_Exec(Task call, int argl, void* args)
132{
133 PCB *curproc, *newproc;
134
135 /* The new process PCB */
136 newproc = acquire_PCB();
137
138 if(newproc == NULL) goto finish; /* We have run out of PIDs! */
139
140 if(get_pid(newproc)<=1) {
141 /* Processes with pid<=1 (the scheduler and the init process)
142 are parentless and are treated specially. */
143 newproc->parent = NULL;
144 }
145 else
146 {
147 /* Inherit parent */
148 curproc = CURPROC;
149
150 /* Add new process to the parent's child list */
151 newproc->parent = curproc;
152 rlist_push_front(& curproc->children_list, & newproc->children_node);
153
154 /* Inherit file streams from parent */
155 for(int i=0; i<MAX_FILEID; i++) {
156 newproc->FIDT[i] = curproc->FIDT[i];
157 if(newproc->FIDT[i])
158 FCB_incref(newproc->FIDT[i]);
159 }
160 }
161
162
163 /* Set the main thread's function */
164 newproc->main_task = call;
165
166 /* Copy the arguments to new storage, owned by the new process */
167 newproc->argl = argl;
168 if(args!=NULL) {
169 newproc->args = malloc(argl);
170 memcpy(newproc->args, args, argl);
171 }
172 else
173 newproc->args=NULL;
174
175 /*
176 Create and wake up the thread for the main function. This must be the last thing
177 we do, because once we wakeup the new thread it may run! so we need to have finished
178 the initialization of the PCB.
179 */
180 if(call != NULL) {
181 newproc->main_thread = spawn_thread(newproc, start_main_thread);
182 wakeup(newproc->main_thread);
183 }
184
185
186finish:
187 return get_pid(newproc);
188}

216static Pid_t wait_for_specific_child(Pid_t cpid, int* status)
217{
218
219 /* Legality checks */
220 if((cpid<0) || (cpid>=MAX_PROC)) {
221 cpid = NOPROC;
222 goto finish;
223 }
224
225 PCB* parent = CURPROC;
226 PCB* child = get_pcb(cpid);
227 if( child == NULL || child->parent != parent)
228 {
229 cpid = NOPROC;
230 goto finish;
231 }
232
233 /* Ok, child is a legal child of mine. Wait for it to exit. */
234 while(child->pstate == ALIVE)
235 kernel_wait(& parent->child_exit, SCHED_USER);
236
237 cleanup_zombie(child, status);
238
239finish:
240 return cpid;
241}

244static Pid_t wait_for_any_child(int* status)
245{
246 Pid_t cpid;
247
248 PCB* parent = CURPROC;
249
250 /* Make sure I have children! */
251 int no_children, has_exited;
252 while(1) {
253 no_children = is_rlist_empty(& parent->children_list);
254 if( no_children ) break;
255
256 has_exited = ! is_rlist_empty(& parent->exited_list);
257 if( has_exited ) break;
258
259 kernel_wait(& parent->child_exit, SCHED_USER);
260 }
261
262 if(no_children)
263 return NOPROC;
264
265 PCB* child = parent->exited_list.next->pcb;
266 assert(child->pstate == ZOMBIE);
267 cpid = get_pid(child);
268 cleanup_zombie(child, status);
269
270 return cpid;
271}

288void sys_Exit(int exitval)
289{
290
291 PCB *curproc = CURPROC; /* cache for efficiency */
292
293 /* First, store the exit status */
294 curproc->exitval = exitval;
295
296 /*
297 Here, we must check that we are not the init task.
298 If we are, we must wait until all child processes exit.
299 */
300 if(get_pid(curproc)==1) {
301
302 while(sys_WaitChild(NOPROC,NULL)!=NOPROC);
303
304 } else {
305
306 /* Reparent any children of the exiting process to the
307 initial task */
308 PCB* initpcb = get_pcb(1);
309 while(!is_rlist_empty(& curproc->children_list)) {
310 rlnode* child = rlist_pop_front(& curproc->children_list);
311 child->pcb->parent = initpcb;
312 rlist_push_front(& initpcb->children_list, child);
313 }
314
315 /* Add exited children to the initial task's exited list
316 and signal the initial task */
317 if(!is_rlist_empty(& curproc->exited_list)) {
318 rlist_append(& initpcb->exited_list, &curproc->exited_list);
319 kernel_broadcast(& initpcb->child_exit);
320 }
321
322 /* Put me into my parent's exited list */
323 rlist_push_front(& curproc->parent->exited_list, &curproc->exited_node);
324 kernel_broadcast(& curproc->parent->child_exit);
325
326 }
327
328 assert(is_rlist_empty(& curproc->children_list));
329 assert(is_rlist_empty(& curproc->exited_list));
330
331
332 /*
333 Do all the other cleanup we want here, close files etc.
334 */
335
336 /* Release the args data */
337 if(curproc->args) {
338 free(curproc->args);
339 curproc->args = NULL;
340 }
341
342 /* Clean up FIDT */
343 for(int i=0;i<MAX_FILEID;i++) {
344 if(curproc->FIDT[i] != NULL) {
345 FCB_decref(curproc->FIDT[i]);
346 curproc->FIDT[i] = NULL;
347 }
348 }
349
350 /* Disconnect my main_thread */
351 curproc->main_thread = NULL;
352
353 /* Now, mark the process as exited. */
354 curproc->pstate = ZOMBIE;
355
356 /* Bye-bye cruel world */
357 kernel_sleep(EXITED, SCHED_USER);
358}
