22/*
23 Pre-emption aware mutex.
24 -------------------------
25
26 This mutex will act as a spinlock if preemption is off, and a
27 yielding mutex if preemption is on.
28
29 Therefore, we can call the same function from both the preemptive and
30 the non-preemptive domain of the kernel.
31
32 The implementation is based on GCC atomics, as the standard C11 primitives
33 are not supported by all recent compilers. Eventually, this will change.
34 */
35void Mutex_Lock(Mutex* lock)
36{
37#define MUTEX_SPINS (cpu_cores()>1 ? 1000 : 10000)
38
39 while(__atomic_test_and_set(lock,__ATOMIC_ACQUIRE)) {
40 int spin=MUTEX_SPINS;
41 while(__atomic_load_n(lock, __ATOMIC_RELAXED)) {
42#if defined(__x86__) || defined(__x86_64__)
43 __builtin_ia32_pause();
44#endif
45 if(spin>0)
46 spin--;
47 else {
48 spin=MUTEX_SPINS;
49 if(cpu_interrupts_enabled())
50 yield(SCHED_MUTEX);
51 }
52 }
53 }
54#undef MUTEX_SPINS
55}

70typedef struct __cv_waiter {
71 rlnode node; /* become part of a ring */
72 TCB* thread; /* thread to wait */
73 sig_atomic_t signalled; /* this is set if the thread is signalled */
74 sig_atomic_t removed; /* this is set if the waiter is removed
75 from the ring */
76} __cv_waiter;
83static inline void remove_from_ring(CondVar* cv, __cv_waiter* w)
84{
85 if(cv->waitset == w) {
86 /* Make cv->waitset safe */
87 __cv_waiter * nextw = w->node.next->obj;
88 cv->waitset = (nextw == w) ? NULL : nextw;
89 }
90 rlist_remove(& w->node);
91}

121static int cv_wait(Mutex* mutex, CondVar* cv,
122 enum SCHED_CAUSE cause, TimerDuration timeout)
123{
124 __cv_waiter waiter = { .thread=cur_thread(), .signalled = 0, .removed=0 };
125 rlnode_init(& waiter.node, &waiter);
126
127 Mutex_Lock(&(cv->waitset_lock));
128 /* We just push the current thread to the back of the list */
129 if(cv->waitset) {
130 __cv_waiter* wset = cv->waitset;
131 rlist_push_back(& wset->node, & waiter.node);
132 } else {
133 cv->waitset = &waiter;
134 }
135
136 /* Now atomically release mutex and sleep */
137 Mutex_Unlock(mutex);
138 sleep_releasing(STOPPED, &(cv->waitset_lock), cause, timeout);
139
140 /* Woke up, we must check wether we were signaled, and tidy up */
141 Mutex_Lock(&(cv->waitset_lock));
142 if(! waiter.removed) {
143 assert(! waiter.signalled);
144
145 /* We must remove ourselves from the ring! */
146 remove_from_ring(cv, &waiter);
147 }
148 Mutex_Unlock(&(cv->waitset_lock));
149
150 Mutex_Lock(mutex);
151 return waiter.signalled;
152}

161static inline void cv_signal(CondVar* cv)
162{
163 /* Wakeup first process in the waiters' queue, if it exists. */
164 while(cv->waitset) {
165 __cv_waiter* waiter = cv->waitset;
166 remove_from_ring(cv, waiter);
167 waiter->removed = 1;
168 if(wakeup(waiter->thread)) {
169 waiter->signalled = 1;
170 return;
171 }
172 }
173}

225/* This mutex is used to implement the kernel semaphore as a monitor. */
226static Mutex kernel_mutex = MUTEX_INIT;
227
228/* Semaphore counter */
229static int kernel_sem = 1;

231/* Semaphore condition */
232static CondVar kernel_sem_cv = COND_INIT;

234void kernel_lock()
235{
236 Mutex_Lock(& kernel_mutex);
237 while(kernel_sem<=0) {
238 Cond_Wait(& kernel_mutex, &kernel_sem_cv);
239 }
240 kernel_sem--;
241 Mutex_Unlock(& kernel_mutex);
242}

244void kernel_unlock()
245{
246 Mutex_Lock(& kernel_mutex);
247 kernel_sem++;
248 Cond_Signal(&kernel_sem_cv);
249 Mutex_Unlock(& kernel_mutex);
250}

252int kernel_wait_wchan(CondVar* cv, enum SCHED_CAUSE cause,
253 const char* wchan_name, TimerDuration timeout)
254{
255 /* Atomically release kernel semaphore */
256 Mutex_Lock(& kernel_mutex);
257 kernel_sem++;
258 Cond_Signal(&kernel_sem_cv);
259
260 int ret = cv_wait(&kernel_mutex, cv, cause, timeout);
261
262 /* Reacquire kernel semaphore */
263 while(kernel_sem<=0)
264 Cond_Wait(& kernel_mutex, &kernel_sem_cv);
265 kernel_sem--;
266 Mutex_Unlock(& kernel_mutex);
267
268 return ret;
269}
