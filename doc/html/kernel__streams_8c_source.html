65int FCB_reserve(size_t num, Fid_t *fid, FCB** fcb)
66{
67 PCB* cur = CURPROC;
68 size_t f=0;
69 uint i;
70
71 /* Find distinct fids */
72 for(i=0; i<num; i++) {
73 while(f<MAX_FILEID && cur->FIDT[f]!=NULL)
74 f++;
75 if(f==MAX_FILEID) break;
76 fid[i] = f; f++;
77 }
78 if(i<num) return 0;
79 /* Allocate FCBs */
80 for(i=0;i<num;i++)
81 if((fcb[i] = acquire_FCB()) == NULL)
82 break;
83 if(i<num) {
84 /* Roll back */
85 while(i>0) {
86 release_FCB(fcb[i-1]);
87 i--;
88 }
89 return 0;
90 }
91 /* Found all */
92 for(i=0;i<num;i++) {
93 cur->FIDT[fid[i]]=fcb[i];
94 FCB_incref(fcb[i]);
95 }
96 return 1;
97}

196int sys_Close(int fd)
197{
198 int retcode = (fd>=0 && fd<MAX_FILEID) ? 0 : -1; /* Closing a closed fd is legal! */
199
200 FCB* fcb = get_fcb(fd);
201
202 if(fcb) {
203 CURPROC->FIDT[fd] = NULL;
204 retcode = FCB_decref(fcb);
205 }

211/*
212 Copy file descriptor oldfd into file descriptor newfd.
213
214 This call returns 0 on success and -1 on failure.
215 Possible reasons for failure:
216 - Either oldfd or newfd is invalid.
217 */
218int sys_Dup2(int oldfd, int newfd)
219{
220 int retcode=0;
221 if(oldfd<0 || newfd<0 || oldfd>=MAX_FILEID || newfd>=MAX_FILEID)
222 return -1;
223
224 FCB* old = get_fcb(oldfd);
225 FCB* new = get_fcb(newfd);
226
227 if(old==NULL) {
228 retcode = -1;
229 }
230 else if(old!=new) {
231 if(new)
232 FCB_decref(new);
233 FCB_incref(old);
234 CURPROC->FIDT[newfd] = old;
235 }
236
237 return retcode;
238}

251Fid_t open_stream(Device_type major, unsigned int minor)
252{
253 Fid_t fid;
254 FCB* fcb;
255
256
257 if(! FCB_reserve(1, &fid, &fcb))
258 goto finerr;
259
260 if(device_open(major, minor, & fcb->streamobj, &fcb->streamfunc)) {
261 FCB_unreserve(1, &fid, &fcb);
262 goto finerr;
263 }
264
265 goto finok;
266finerr:
267 fid = NOFILE;
268finok:
269 return fid;
270}
