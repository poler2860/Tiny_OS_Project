95static int exec_wrapper(int argl, void* args)
96{
97 /* unpack the program pointer */
98 Program prog;
99
100 /* unpack the prog pointer */
101 memcpy(&prog, args, sizeof(prog));
102
103 argl -= sizeof(prog);
104 args += sizeof(prog);
105
106 /* unpack the string vector */
107 size_t argc = argscount(argl, args);
108 const char* argv[argc];
109 argvunpack(argc, argv, argl, args);
110
111 /* Make the call */
112 return prog(argc, argv);
113}

116int ParseProcInfo(procinfo* pinfo, Program* prog, int argc, const char** argv )
117{
118 if(pinfo->main_task != exec_wrapper)
119 /* We do not recognize the format! */
120 return -1;
121
122 if(pinfo->argl > PROCINFO_MAX_ARGS_SIZE)
123 /* The full argument is not available */
124 return -1;
125
126 int argl = pinfo->argl;
127 void* args = pinfo->args;
128
129 /* unpack the program pointer */
130 if(prog) memcpy(&prog, args, sizeof(prog));
131
132 argl -= sizeof(Program*);
133 args += sizeof(Program*);
134
135 /* unpack the string vector */
136 size_t N = argscount(argl, args);
137 if(argv) {
138 if(argc>N)
139 argc = N;
140 argvunpack(argc, argv, argl, args);
141 }
142
143 return N;
144}

148int Execute(Program prog, size_t argc, const char** argv)
149{
150 /* We will pack the prog pointer and the arguments to
151 an argument buffer.
152 */
153
154 /* compute the argument buffer size */
155 size_t argl = argvlen(argc, argv) + sizeof(prog);
156
157 /* allocate the buffer */
158 char args[argl];
159
160 /* put the pointer at the start */
161 memcpy(args, &prog, sizeof(prog));
162
163 /* add the string vector */
164 argvpack(args+sizeof(prog), argc, argv);
165
166 /* Execute the process */
167 return Exec(exec_wrapper, argl, args);
168}

172void BarrierSync(barrier* bar, unsigned int n)
173{
174 assert(n>0);
175 Mutex_Lock(& bar->mx);
176
177 int epoch = bar->epoch;
178 assert(bar->count < n);
179
180 bar->count ++;
181 if(bar->count == n) {
182 bar->epoch ++;
183 bar->count = 0;
184 Cond_Broadcast(&bar->cv);
185 }
186
187 while(epoch == bar->epoch)
188 Cond_Wait(&bar->mx, &bar->cv);
189
190 Mutex_Unlock(& bar->mx);
191}
